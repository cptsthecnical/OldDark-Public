# Se trata de desactivar el inicio de sesión automático en Windows

# modifico las variables
Botón Windows + R
regedit
ruta : HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Winlogon

# dejo los siguientes registros
- AutoAdminLogon = 0                        # desactiva el inicio automático
- DefaultPassword → eliminar                # elimina la contraseña almacenada
- DefaultUserName → dejar igual             # no afecta seguridad

# compruebo con powershell
$rk = 'HKLM:\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Winlogon'
Get-ItemProperty -Path $rk | Select-Object AutoAdminLogon, DefaultUserName, DefaultPassword
